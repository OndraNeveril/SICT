\section*{Hardware}
Systémový -- systémová jednotka (počítačová skříň)

Periferie -- Vstupní/výstupní zařízení (tiskárna, myš, klávesnice ...)

\subsection*{Systémová jednotka}
\begin{itemize}
\item Zdroj (do zásuvky, UPS -- záložní zdroj napájení)
\item Základní deska
\item Disky
\item CPU
\item RAM
\end{itemize}

\subsection*{Základní deska}
\begin{itemize}
\item Porty (vstupy, výstupy)
\item Patice na CPU
\item Chipset
\item Patice pro SATA disky
\item Konektor napájení
\item Sloty na přídavné disky
\item BIOS/UEFI (BIOS -- starší firmware na základní desce. UEFI -- modernější, grafické, rychlejší, bezchybnější, snadnější aktualizace) -- zajišťují test HW, zavedení OS
\end{itemize}

\subsection*{Porty}
\begin{itemize}
\item Napájení
\item PS/2 fialový -- klávesnice, zelený -- myš
\item COM1 -- sériový port
\item LPT1 -- paralelní (rychlejší -- tiskárny)
\item USB (Universal Serial Bus)
\item RJ45 -- LAN (Ethernet)
\item DVI
\item HDMI
\item ...
\end{itemize}

\subsection*{USB standardy}
\begin{itemize}
\item USB $1.0$ -- $12 Mb/s$
\item USB $2.0$ -- $480 Mb/s$
\item USB $3.0$ -- $5 Gb/s$
\item USB $3.1$ -- $10 Gb/s$
\item USB $4.0$ -- $80 Gb/s$
\end{itemize}

\section*{Tiskárny}

\begin{enumerate}
\item Jehličkové -- drahé, hlučné, malé DPI, levný tisk, pouze černě
\item Laserové a LED -- černý i barevný tisk, kvalitní, čistá výměna tonerů,  vysoké provozní náklady.
\item Inkoustové tiskárny - barevný tisk, kvalitní, rychlé, levné
\item termální tiskárny $\cdots$
\item $3$D tiskárny
\end{enumerate}

Rozdělení:
\begin{itemize}
\item Tiskárny
\item Multifunkce - scanner - scanování, kopírování
\item (Fax)
\end{itemize}

Připojení
\begin{itemize}
\item USB
\item LAN -- výhodné pro více PC v síti
\item Wifi
\item Bluetooth
\end{itemize}
